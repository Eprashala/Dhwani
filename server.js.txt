// server.js
import express from 'express';
import bodyParser from 'body-parser';
import dotenv from 'dotenv';
import fs from 'fs';
import OpenAI from 'openai';

dotenv.config();
const app = express();
app.use(bodyParser.json({limit: '2mb'}));
app.use(express.static('public')); // optional

const client = new OpenAI({ apiKey: process.env.OPENAI_API_KEY });

// POST /chat
app.post('/chat', async (req, res) => {
  try {
    const { messages } = req.body; // messages: [{role,content},...]
    if(!messages) return res.status(400).json({error:'no messages'});
    // call Chat Completions (Responses API or chat/completions).
    // We'll use chat.completions style (modern clients support responses too).
    const completion = await client.chat.completions.create({
      model: 'gpt-4o-mini', // choose your available chat model
      messages,
      max_tokens: 250
    });
    const assistantText = completion.choices?.[0]?.message?.content ?? 'क्षमस्व — उत्तर मिळाले नाही.';
    res.json({text: assistantText});
  } catch(err){
    console.error(err);
    res.status(500).json({error: String(err)});
  }
});

// POST /tts
app.post('/tts', async (req, res) => {
  try {
    const { text } = req.body;
    if(!text) return res.status(400).json({error:'no text'});
    // Use the OpenAI TTS endpoint: client.audio.speech.create or similar.
    // Depending on your SDK version the call may differ; below is a common pattern.
    const mp3 = await client.audio.speech.create({
      model: 'gpt-4o-mini-tts', // or 'tts-1' depending on availability
      voice: 'alloy', // pick a voice your account supports
      input: text,
      response_format: 'mp3'
    });
    // mp3 is a web stream/response — convert to buffer
    const buffer = Buffer.from(await mp3.arrayBuffer());
    res.set('Content-Type', 'audio/mpeg');
    res.send(buffer);
  } catch(err){
    console.error('tts error', err);
    res.status(500).json({error: String(err)});
  }
});

const port = process.env.PORT || 3000;
app.listen(port, () => console.log('Server running on', port));
